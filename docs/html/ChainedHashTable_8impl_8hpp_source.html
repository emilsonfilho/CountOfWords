82void ChainedHashTable<Key, Value, Hash>::insert(const Key& key, const Value& value) {
83 this->checkAndRehash();
84
85 size_t slot = hashCode(key);
86
87 if (!this->table[slot].empty())
88 this->incrementCollisionsCount();
89
90 for (const auto& p : this->table[slot]) {
91 this->comparisonsCount++;
